<script>
  let messages = [];
  let source = 'puter';
  let orKey = localStorage.getItem('aura_or_key') || '';
  let sysPrompt = localStorage.getItem('aura_sys') || '';
  let activeModel = localStorage.getItem('aura_model') || 'auto';
  let isLoading = false;

  const PUTER_MODELS = [
    { id:'auto', name:'Auto', desc:'Best available · Puter picks', color:'#a78bfa' },
    { id:'gpt-4o-mini', name:'GPT-4o Mini', desc:'Fast & efficient · OpenAI', color:'#60a5fa' },
    { id:'gpt-4o', name:'GPT-4o', desc:'Most capable · OpenAI', color:'#34d399' },
    { id:'claude-sonnet-4-5', name:'Claude Sonnet', desc:'Smart reasoning · Anthropic', color:'#f472b6' },
    { id:'o1-mini', name:'o1 Mini', desc:'Deep reasoning · OpenAI', color:'#fb923c' },
  ];

  const OR_MODELS = [
    { id:'deepseek/deepseek-r1:free', name:'DeepSeek R1', desc:'Free · Best reasoning', color:'#60a5fa' },
    { id:'google/gemini-2.0-flash-lite-preview-02-05:free', name:'Gemini 2.0 Flash', desc:'Free · Fast & smart', color:'#34d399' },
    { id:'meta-llama/llama-3.3-70b-instruct:free', name:'Llama 3.3 70B', desc:'Free · Meta', color:'#fb923c' },
    { id:'mistralai/mistral-7b-instruct:free', name:'Mistral 7B', desc:'Free · Lightweight', color:'#a78bfa' },
    { id:'qwen/qwen2.5-72b-instruct:free', name:'Qwen 2.5 72B', desc:'Free · Alibaba', color:'#f472b6' },
  ];

  window.addEventListener('DOMContentLoaded', () => {
    updateGreeting(); updateDateCard(); renderModelList();
    const inp = document.getElementById('input');
    inp.addEventListener('input', () => { document.getElementById('send-btn').disabled = !inp.value.trim(); });
    inp.addEventListener('keydown', e => { if(e.key==='Enter'&&!e.shiftKey){e.preventDefault();send();} });
    if(orKey) document.getElementById('or-key-input').value = orKey;
    if(sysPrompt) document.getElementById('sys-prompt').value = sysPrompt;
    updateLabel();
  });

  function updateGreeting() {
    const h = new Date().getHours();
    const el = document.getElementById('time-label');
    el.textContent = h<12?'🌤 Good Morning':h<17?'☀️ Good Afternoon':'🌙 Good Evening';
  }

  function updateDateCard() {
    const d = new Date();
    const days=['Sun','Mon','Tue','Wed','Thu','Fri','Sat'];
    const months=['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
    const dc = document.getElementById('day-card');
    if(dc) { dc.childNodes[0].textContent = days[d.getDay()]; }
    const dateEl = document.getElementById('date-card');
    if(dateEl) dateEl.textContent = months[d.getMonth()]+' '+d.getDate();
  }

  function renderModelList() {
    const models = source==='puter'?PUTER_MODELS:OR_MODELS;
    document.getElementById('model-list').innerHTML = models.map(m=>`
      <div class="model-item ${activeModel===m.id?'active':''}" onclick="selectModel('${m.id}')">
        <div class="model-dot" style="background:${m.color}"></div>
        <div style="flex:1"><div class="model-name">${m.name}</div><div class="model-desc">${m.desc}</div></div>
        <svg class="model-check" width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="3"><polyline points="20 6 9 17 4 12"/></svg>
      </div>`).join('');
  }

  function setSource(s) {
    source = s;
    document.getElementById('src-puter').classList.toggle('active',s==='puter');
    document.getElementById('src-openrouter').classList.toggle('active',s==='openrouter');
    document.getElementById('key-section').classList.toggle('show',s==='openrouter');
    activeModel = s==='puter'?'auto':OR_MODELS[0].id;
    renderModelList(); updateLabel();
  }

  function selectModel(id) {
    activeModel = id;
    localStorage.setItem('aura_model', id);
    renderModelList(); updateLabel(); closeSheets();
  }

  function saveKey() {
    orKey = document.getElementById('or-key-input').value.trim();
    localStorage.setItem('aura_or_key', orKey);
    closeSheets(); addSysMsg('✅ OpenRouter key saved');
  }

  function updateLabel() {
    const models = source==='puter'?PUTER_MODELS:OR_MODELS;
    const m = models.find(x=>x.id===activeModel)||models[0];
    document.getElementById('active-label').textContent = `${m.name} (${source==='puter'?'Puter':'OpenRouter'})`;
  }

  function saveSysPrompt() {
    sysPrompt = document.getElementById('sys-prompt').value;
    localStorage.setItem('aura_sys', sysPrompt);
    closeSheets(); addSysMsg('✅ Settings saved');
  }

  async function send() {
    const inp = document.getElementById('input');
    const text = inp.value.trim();
    if(!text||isLoading) return;
    inp.value=''; autoResize(inp);
    document.getElementById('send-btn').disabled = true;
    hideWelcome();
    addBubble(text,'user');
    messages.push({role:'user',content:text});
    await getResponse();
  }

  async function quickSend(text) {
    hideWelcome();
    addBubble(text,'user');
    messages.push({role:'user',content:text});
    await getResponse();
  }

  async function getResponse() {
    isLoading = true;
    const typing = addTyping();
    try {
      const full = sysPrompt?[{role:'system',content:sysPrompt},...messages]:messages;
      let reply='';
      if(source==='puter') {
        const opts = activeModel!=='auto'?{model:activeModel}:{};
        const res = await puter.ai.chat(full, opts);
        reply = res?.message?.content?.[0]?.text || res?.toString() || 'No response';
      } else {
        if(!orKey){ reply='⚠️ No OpenRouter key. Tap model icon to add one.'; }
        else {
          const res = await fetch('https://openrouter.ai/api/v1/chat/completions',{
            method:'POST',
            headers:{'Authorization':`Bearer ${orKey}`,'Content-Type':'application/json'},
            body:JSON.stringify({model:activeModel,messages:full})
          });
          const d = await res.json();
          reply = d.error?('⚠️ '+(d.error.message||'API Error')):d.choices?.[0]?.message?.content||'No response';
        }
      }
      typing.remove();
      addBubble(reply,'ai');
      messages.push({role:'assistant',content:reply});
    } catch(e) {
      typing.remove();
      addBubble('⚠️ '+e.message,'ai');
    }
    isLoading=false;
    document.getElementById('send-btn').disabled=false;
  }

  function addBubble(text, role) {
    const wrap = document.createElement('div');
    wrap.className = `msg-row ${role}`;
    if(role==='ai'){const n=document.createElement('div');n.className='ai-name';n.textContent='Aura';wrap.appendChild(n);}
    const b = document.createElement('div');
    b.className='bubble'; b.textContent=text;
    wrap.appendChild(b);
    document.getElementById('messages').appendChild(wrap);
    wrap.scrollIntoView({behavior:'smooth',block:'end'});
    return wrap;
  }

  function addTyping() {
    const wrap = document.createElement('div');
    wrap.className='msg-row ai';
    const n=document.createElement('div');n.className='ai-name';n.textContent='Aura';
    const b=document.createElement('div');b.className='bubble';
    b.innerHTML='<div class="typing-dots"><span></span><span></span><span></span></div>';
    wrap.appendChild(n);wrap.appendChild(b);
    document.getElementById('messages').appendChild(wrap);
    wrap.scrollIntoView({behavior:'smooth',block:'end'});
    return wrap;
  }

  function addSysMsg(text) {
    const d=document.createElement('div');
    d.className='sys-msg';d.textContent=text;
    document.getElementById('messages').appendChild(d);
  }

  function hideWelcome() {
    const w=document.getElementById('welcome');
    if(w){w.style.opacity='0';setTimeout(()=>w.remove(),300);}
  }

  function clearChat() {
    messages=[];
    document.getElementById('messages').innerHTML=`
    <div id="welcome">
      <div class="welcome-time" id="time-label"></div>
